library(tidyverse)

all_vio_descr = read_csv("./data/inspection_sub_latest_date.csv") %>% 
  janitor::clean_names()%>% 
   mutate(
    violation_description = as.factor(violation_description)
  ) %>%
  group_by(violation_description) %>%
  summarize(
    n = n()
  ) %>%
  mutate(
    rank = rank(-n, ties.method = "first")
  ) %>%
  arrange(desc(n))
